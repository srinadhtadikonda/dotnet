using System;
using System.Collections.Generic;
using System.Linq;

namespace WebApp
{
    public partial class MultiSelect : System.Web.UI.Page
    {
        protected void lstItems_SelectedIndexChanged(object sender, EventArgs e)
        {
            var selectedValues = lstItems.Items
                                         .Cast<System.Web.UI.WebControls.ListItem>()
                                         .Where(item => item.Selected)
                                         .Select(item => item.Value)
                                         .ToList();

            if (selectedValues.Count > 0)
            {
                lblResult.Text = "Selected values: " + string.Join(", ", selectedValues);
            }
            else
            {
                lblResult.Text = "No values selected.";
            }
        }
    }
}
