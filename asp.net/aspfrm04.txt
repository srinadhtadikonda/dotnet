using System;
using System.Collections.Generic;

namespace CoffeeApp
{
    public partial class CoffeeOrder : System.Web.UI.Page
    {
        protected void UpdateOrder(object sender, EventArgs e)
        {
            List<string> selections = new List<string>();

            if (chkCream.Checked)
                selections.Add("cream");

            if (chkSugar.Checked)
                selections.Add("sugar");

            if (selections.Count > 0)
                txtOrder.Text = "You ordered coffee with " + string.Join(", ", selections);
            else
                txtOrder.Text = "";
        }
    }
}
