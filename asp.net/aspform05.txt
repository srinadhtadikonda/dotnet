using System;

namespace CoffeeApp
{
    public partial class CoffeeOrder : System.Web.UI.Page
    {
        protected void UpdateOrder(object sender, EventArgs e)
        {
            if (rbCream.Checked)
            {
                txtOrder.Text = "You ordered coffee with cream";
            }
            else if (rbSugar.Checked)
            {
                txtOrder.Text = "You ordered coffee with sugar";
            }
            else
            {
                txtOrder.Text = "";
            }
        }
    }
}
